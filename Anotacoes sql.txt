CREATE TABLE Matriculas(
     id int AUTO_INCREMENT PRIMARY KEY,
     id_aluno INT,
     id_professor int,
     data_matricula DATE,
     FOREIGN KEY (id_aluno) REFERENCES alunos (id),
     FOREIGN KEY (id_professor) REFERENCES professores (id)
